HDU 1020 Eecoding

2023-07-31

HDU Encoding

题目大意

将输入的大写字母字符串转换格式输出，在大写字母前输出该字母的个数，如果个数为1，则可忽略

代码

#include<iostream>
#include<string>
using namespace std;

char s[10005];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>s;
		int len=strlen(s);
		int cnt=1;
		for(int i=0;i<len;i++)
		{
			if(s[i]==s[i+1])
				cnt++;
			else
			{
				if(cnt==1)
					cout<<s[i];
				else
					cout<<cnt<<s[i];
				cnt=1;
			}
		}
		cout<<endl;
	}
	return 0;
}
